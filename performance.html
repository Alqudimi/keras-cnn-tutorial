تحسين الأداء (Performance Optimization)

تقنيات لتسريع التدريب، تقليل استخدام الذاكرة، والاستفادة القصوى من الأجهزة.

مقدمة

تدريب نماذج التعلم العميق قد يستغرق ساعات أو أياماً. تحسين الأداء يوفر الوقت والموارد ويسمح بتجربة المزيد من الأفكار في وقت أقل.

المكاسب المتوقعة

باستخدام التقنيات في هذا الدليل، يمكنك تقليل وقت التدريب بنسبة 50-70% واستخدام batch sizes أكبر بـ 2-3 مرات!

تدريب أسرع

1. تحسين Data Pipeline

Python 
import tensorflow as tf

def create_optimized_dataset(file_pattern, batch_size, img_size=(224, 224)):
    """
    إنشاء data pipeline محسن
    """
    AUTOTUNE = tf.data.AUTOTUNE
    
    # قراءة الملفات
    dataset = tf.data.Dataset.list_files(file_pattern, shuffle=True)
    
    # معالجة الصور بشكل متوازي
    def load_and_preprocess(file_path):
        img = tf.io.read_file(file_path)
        img = tf.image.decode_jpeg(img, channels=3)
        img = tf.image.resize(img, img_size)
        img = img / 255.0
        
        # استخراج التسمية من المسار
        parts = tf.strings.split(file_path, '/')
        label = parts[-2]  # اسم المجلد = الفئة
        
        return img, label
    
    dataset = dataset.map(load_and_preprocess, num_parallel_calls=AUTOTUNE)
    
    # التحسينات الأساسية
    dataset = dataset.cache()           # تخزين في الذاكرة
    dataset = dataset.shuffle(1000)     # خلط البيانات
    dataset = dataset.batch(batch_size)
    dataset = dataset.prefetch(AUTOTUNE)  # تحميل مسبق
    
    return dataset

# استخدام
train_dataset = create_optimized_dataset(
    'dataset/train/*/*.jpg',
    batch_size=32
)

2. استخدام TFRecord

Python 
# إنشاء TFRecord (أسرع 2-3x من قراءة الملفات)
def create_tfrecord(image_paths, labels, output_path):
    with tf.io.TFRecordWriter(output_path) as writer:
        for img_path, label in zip(image_paths, labels):
            img = tf.io.read_file(img_path)
            
            feature = {
                'image': tf.train.Feature(bytes_list=tf.train.BytesList(value=[img.numpy()])),
                'label': tf.train.Feature(int64_list=tf.train.Int64List(value=[label]))
            }
            
            example = tf.train.Example(features=tf.train.Features(feature=feature))
            writer.write(example.SerializeToString())

# قراءة TFRecord
def parse_tfrecord(example):
    feature_description = {
        'image': tf.io.FixedLenFeature([], tf.string),
        'label': tf.io.FixedLenFeature([], tf.int64)
    }
    
    parsed = tf.io.parse_single_example(example, feature_description)
    image = tf.image.decode_jpeg(parsed['image'], channels=3)
    image = tf.image.resize(image, [224, 224]) / 255.0
    
    return image, parsed['label']

# استخدام
dataset = tf.data.TFRecordDataset('train.tfrecord')
dataset = dataset.map(parse_tfrecord, num_parallel_calls=tf.data.AUTOTUNE)
dataset = dataset.batch(32).prefetch(tf.data.AUTOTUNE)

Mixed Precision Training

استخدام أرقام 16-bit بدلاً من 32-bit يسرع التدريب بنسبة 2-3x ويقلل استخدام الذاكرة بنسبة 50%.

Python 
import tensorflow as tf
from tensorflow.keras import mixed_precision

# تفعيل Mixed Precision - سطر واحد فقط!
mixed_precision.set_global_policy('mixed_float16')

# بناء النموذج كالمعتاد
model = tf.keras.Sequential([
    tf.keras.layers.Conv2D(32, (3, 3), activation='relu'),
    tf.keras.layers.BatchNormalization(),
    tf.keras.layers.MaxPooling2D(),
    # ... المزيد من الطبقات
    
    # مهم: الطبقة الأخيرة يجب أن تكون float32
    tf.keras.layers.Dense(num_classes, dtype='float32')
])

# التحقق من السياسة
print(f"Compute dtype: {mixed_precision.global_policy().compute_dtype}")
print(f"Variable dtype: {mixed_precision.global_policy().variable_dtype}")

# تجميع وتدريب كالمعتاد

    optimizer='adam',
    loss='sparse_categorical_crossentropy',

ملاحظة مهمة

Mixed Precision يتطلب GPU حديث (NVIDIA Volta أو أحدث). على GPU قديم، قد لا تحصل على تسريع.

Batch Size Tuning

اختيار Batch Size المناسب يؤثر على سرعة التدريب وجودة النتائج واستخدام الذاكرة.

Batch Size المزايا العيوب
صغير (8-16) أقل ذاكرة، تعميم أفضل أبطأ، تذبذب أكثر
متوسط (32-64) توازن جيد قد لا يستغل GPU بالكامل
كبير (128+) أسرع، استغلال GPU أفضل ذاكرة أكثر، قد يحتاج LR أعلى
Python 
# البحث عن أكبر batch size ممكن
def find_max_batch_size(model, input_shape, start=32, max_size=512):
    batch_size = start
    
    while batch_size <= max_size:
        try:
            # إنشاء بيانات وهمية
            dummy_data = tf.random.normal([batch_size] + list(input_shape))
            
            # محاولة forward و backward pass
            with tf.GradientTape() as tape:
                output = model(dummy_data, training=True)
                loss = tf.reduce_mean(output)
            
            grads = tape.gradient(loss, model.trainable_variables)
            
            print(f"Batch size {batch_size}: OK")
            batch_size *= 2
            
            # تنظيف الذاكرة
            del dummy_data, output, loss, grads
            tf.keras.backend.clear_session()
            
        except tf.errors.ResourceExhaustedError:
            print(f"Batch size {batch_size}: Out of Memory")
            return batch_size // 2
    
    return batch_size // 2

max_batch = find_max_batch_size(model, (224, 224, 3))
print(f"Maximum batch size: {max_batch}")

# قاعدة Linear Scaling Rule
# عند زيادة batch size، زِد learning rate بنفس النسبة
base_lr = 0.001
base_batch_size = 32
new_batch_size = 128

new_lr = base_lr * (new_batch_size / base_batch_size)
print(f"Adjusted LR: {new_lr}")  # 0.004
