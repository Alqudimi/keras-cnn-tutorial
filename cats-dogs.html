مشروع تصنيف القطط والكلاب (Cats vs Dogs)

مشروع عملي كامل لبناء نموذج CNN من الصفر لتصنيف صور القطط والكلاب. يغطي المشروع جميع المراحل من تحضير البيانات إلى نشر النموذج.

مقدمة المشروع

مشروع تصنيف القطط والكلاب هو أحد أشهر مشاريع التعلم العميق للمبتدئين. يُعد هذا المشروع نقطة انطلاق ممتازة لفهم كيفية بناء شبكة عصبية تلافيفية (CNN) من الصفر وتدريبها على مجموعة بيانات حقيقية.

عن مجموعة البيانات

مجموعة بيانات Cats vs Dogs من Kaggle تحتوي على حوالي 25,000 صورة مقسمة بالتساوي بين القطط والكلاب. هذا التوازن يجعلها مثالية للتدريب دون الحاجة لتقنيات معالجة عدم التوازن.

لماذا هذا المشروع؟

  • التصنيف الثنائي: أبسط أنواع التصنيف - فئتان فقط
  • بيانات وفيرة: 25,000 صورة كافية للتدريب الجيد
  • تحديات واقعية: صور بأحجام وزوايا وخلفيات مختلفة
  • نتائج ملموسة: يمكن اختبار النموذج على صور حقيقية من الإنترنت

كود تقسيم البيانات

Python 
import os
import shutil
from sklearn.model_selection import train_test_split

def organize_dataset(source_dir, dest_dir):
    """
    تنظيم البيانات في مجلدات train/val/test
    """
    categories = ['cats', 'dogs']
    
    for category in categories:
        # جمع جميع الصور
        all_images = os.listdir(os.path.join(source_dir, category))
        
        # تقسيم: 80% تدريب، 10% تحقق، 10% اختبار
        train_imgs, temp_imgs = train_test_split(all_images, test_size=0.2, random_state=42)
        val_imgs, test_imgs = train_test_split(temp_imgs, test_size=0.5, random_state=42)
        
        # إنشاء المجلدات ونقل الصور
        for split, images in [('train', train_imgs), ('validation', val_imgs), ('test', test_imgs)]:
            split_dir = os.path.join(dest_dir, split, category)
            os.makedirs(split_dir, exist_ok=True)
            
            for img in images:
                src = os.path.join(source_dir, category, img)
                dst = os.path.join(split_dir, img)
                shutil.copy(src, dst)
        
        print(f"{category}: Train={len(train_imgs)}, Val={len(val_imgs)}, Test={len(test_imgs)}")

تنظيف الصور التالفة

قبل التدريب، يجب فحص جميع الصور وإزالة التالف منها. الصور التالفة قد تسبب أخطاء أثناء التدريب أو تؤثر سلبًا على أداء النموذج.

أنواع الصور التالفة

قد تجد صورًا بحجم 0 بايت، صور لا يمكن فتحها، صور بصيغ غير مدعومة، أو صور مقطوعة بشكل غير كامل. هذا الكود يكتشف كل هذه الحالات.

Python 
import os
from PIL import Image
import tensorflow as tf

def check_and_remove_corrupted_images(directory):
    """
    فحص وإزالة الصور التالفة من المجلد
    """
    corrupted_count = 0
    valid_extensions = {'.jpg', '.jpeg', '.png', '.bmp', '.gif'}
    
    for root, dirs, files in os.walk(directory):
        for filename in files:
            filepath = os.path.join(root, filename)
            
            # التحقق من الامتداد
            ext = os.path.splitext(filename)[1].lower()
            if ext not in valid_extensions:
                print(f"امتداد غير مدعوم: {filepath}")
                os.remove(filepath)
                corrupted_count += 1
                continue
            
            try:
                # محاولة فتح الصورة باستخدام PIL
                with Image.open(filepath) as img:
                    img.verify()
                
                # محاولة قراءة الصورة باستخدام TensorFlow
                img_bytes = tf.io.read_file(filepath)
                tf.image.decode_image(img_bytes)
                
            except Exception as e:
                print(f"صورة تالفة: {filepath} - {str(e)}")
                os.remove(filepath)
                corrupted_count += 1
    
    print(f"\nإجمالي الصور التالفة المحذوفة: {corrupted_count}")
    return corrupted_count

# تشغيل الفحص
check_and_remove_corrupted_images('dataset/')

زيادة البيانات (Data Augmentation)

زيادة البيانات تساعد النموذج على التعميم بشكل أفضل عن طريق إنشاء تحويلات مختلفة للصور الأصلية. سنستخدم تحويلات بسيطة مناسبة لهذا المشروع.

Python 
from tensorflow.keras.preprocessing.image import ImageDataGenerator

# إعدادات زيادة البيانات للتدريب
train_datagen = ImageDataGenerator(
    rescale=1./255,              # تطبيع القيم من 0-255 إلى 0-1
    rotation_range=20,           # دوران عشوائي حتى 20 درجة
    width_shift_range=0.2,       # إزاحة أفقية عشوائية
    height_shift_range=0.2,      # إزاحة رأسية عشوائية
    shear_range=0.2,             # قص عشوائي
    zoom_range=0.2,              # تكبير/تصغير عشوائي
    horizontal_flip=True,        # انعكاس أفقي
    fill_mode='nearest'          # ملء الفراغات بأقرب قيمة
)

# للتحقق والاختبار: تطبيع فقط بدون augmentation
val_datagen = ImageDataGenerator(rescale=1./255)
test_datagen = ImageDataGenerator(rescale=1./255)

# إنشاء المولدات
IMG_SIZE = (150, 150)
BATCH_SIZE = 32

train_generator = train_datagen.flow_from_directory(
    'dataset/train',
    target_size=IMG_SIZE,
    batch_size=BATCH_SIZE,
    class_mode='binary'  # تصنيف ثنائي
)

val_generator = val_datagen.flow_from_directory(
    'dataset/validation',
    target_size=IMG_SIZE,
    batch_size=BATCH_SIZE,
    class_mode='binary'
)

test_generator = test_datagen.flow_from_directory(
    'dataset/test',
    target_size=IMG_SIZE,
    batch_size=BATCH_SIZE,
    class_mode='binary',
    shuffle=False  # مهم للتقييم الصحيح
)

بناء نموذج CNN يدوي بالكامل

سنبني شبكة CNN من الصفر تتكون من عدة كتل تلافيفية متتالية. كل كتلة تحتوي على: Conv2D → BatchNormalization → ReLU → MaxPooling.

Python 
from tensorflow.keras import Sequential
from tensorflow.keras.layers import (
    Conv2D, BatchNormalization, Activation, MaxPooling2D,
    Flatten, Dense, Dropout
)

def build_cats_dogs_model(input_shape=(150, 150, 3)):
    """
    بناء نموذج CNN لتصنيف القطط والكلاب
    """
    model = Sequential([
        # ========== الكتلة الأولى ==========
        Conv2D(32, (3, 3), padding='same', input_shape=input_shape),
        BatchNormalization(),
        Activation('relu'),
        MaxPooling2D((2, 2)),
        
        # ========== الكتلة الثانية ==========
        Conv2D(64, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        MaxPooling2D((2, 2)),
        
        # ========== الكتلة الثالثة ==========
        Conv2D(128, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        MaxPooling2D((2, 2)),
        
        # ========== الكتلة الرابعة ==========
        Conv2D(256, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        MaxPooling2D((2, 2)),
        
        # ========== طبقات التصنيف ==========
        Flatten(),
        Dense(512, activation='relu'),
        Dropout(0.5),
        Dense(1, activation='sigmoid')  # خرج واحد للتصنيف الثنائي
    ])
    
    return model

# إنشاء النموذج
model = build_cats_dogs_model()
model.summary()
الطبقة الشكل الناتج عدد المعاملات الوصف
Input (150, 150, 3) 0 صورة RGB بحجم 150×150
Conv2D + BN + ReLU (150, 150, 32) ~1,000 32 فلتر 3×3
MaxPool (75, 75, 32) 0 تقليص الأبعاد للنصف
Conv2D + BN + ReLU (75, 75, 64) ~18,500 64 فلتر 3×3
MaxPool (37, 37, 64) 0 تقليص الأبعاد
Dense(512) (512) ~4M طبقة كثيفة
Dense(1, sigmoid) (1) 513 خرج ثنائي

إعدادات التدريب (Optimizer – Loss – Callbacks)

إعدادات التدريب الصحيحة ضرورية للحصول على نموذج جيد. سنستخدم Adam كمحسن مع Binary Crossentropy كدالة خسارة للتصنيف الثنائي.

Python 
from tensorflow.keras.optimizers import Adam
from tensorflow.keras.callbacks import (
    EarlyStopping, ModelCheckpoint, ReduceLROnPlateau, TensorBoard
)
import datetime

# تجميع النموذج
model.compile(
    optimizer=Adam(learning_rate=0.001),
    loss='binary_crossentropy',
    metrics=['accuracy']
)

# إعداد الـ Callbacks
callbacks = [
    # الإيقاف المبكر عند توقف التحسن
    EarlyStopping(
        monitor='val_loss',
        patience=10,
        restore_best_weights=True,
        verbose=1
    ),
    
    # حفظ أفضل نموذج
    ModelCheckpoint(
        'best_cats_dogs_model.keras',
        monitor='val_accuracy',
        save_best_only=True,
        verbose=1
    ),
    
    # تقليل معدل التعلم عند الثبات
    ReduceLROnPlateau(
        monitor='val_loss',
        factor=0.5,
        patience=5,
        min_lr=1e-7,
        verbose=1
    ),
    
    # تسجيل للـ TensorBoard
    TensorBoard(
        log_dir=f'logs/{datetime.datetime.now().strftime("%Y%m%d-%H%M%S")}',
        histogram_freq=1
    )
]

# التدريب
EPOCHS = 50

history = model.fit(
    train_generator,
    epochs=EPOCHS,
    validation_data=val_generator,
    callbacks=callbacks,
    verbose=1
)

لماذا هذه الإعدادات؟

  • Adam: محسن قوي يتكيف تلقائيًا مع معدلات التعلم
  • Binary Crossentropy: مثالي للتصنيف الثنائي
  • EarlyStopping: يمنع الإفراط في التدريب
  • ReduceLROnPlateau: يحسن التقارب في المراحل المتأخرة

رسم منحنيات Accuracy و Loss

رسم منحنيات التدريب يساعد في فهم سلوك النموذج واكتشاف مشاكل مثل الإفراط في التكييف (Overfitting).

Python 
import matplotlib.pyplot as plt

def plot_training_history(history):
    """
    رسم منحنيات التدريب: Accuracy و Loss
    """
    fig, axes = plt.subplots(1, 2, figsize=(14, 5))
    
    # منحنى الدقة (Accuracy)
    axes[0].plot(history.history['accuracy'], label='Training Accuracy', linewidth=2)
    axes[0].plot(history.history['val_accuracy'], label='Validation Accuracy', linewidth=2)
    axes[0].set_title('Model Accuracy', fontsize=14, fontweight='bold')
    axes[0].set_xlabel('Epoch')
    axes[0].set_ylabel('Accuracy')
    axes[0].legend(loc='lower right')
    axes[0].grid(True, alpha=0.3)
    axes[0].set_ylim([0, 1])
    
    # منحنى الخسارة (Loss)
    axes[1].plot(history.history['loss'], label='Training Loss', linewidth=2)
    axes[1].plot(history.history['val_loss'], label='Validation Loss', linewidth=2)
    axes[1].set_title('Model Loss', fontsize=14, fontweight='bold')
    axes[1].set_xlabel('Epoch')
    axes[1].set_ylabel('Loss')
    axes[1].legend(loc='upper right')
    axes[1].grid(True, alpha=0.3)
    
    plt.tight_layout()
    plt.savefig('training_curves.png', dpi=150, bbox_inches='tight')
    plt.show()

# رسم المنحنيات
plot_training_history(history)

اختبار النموذج على صور حقيقية

اختبار النموذج على صور جديدة من خارج مجموعة البيانات يعطي صورة حقيقية عن أداء النموذج في العالم الحقيقي.

Python 
from tensorflow.keras.preprocessing import image
import requests
from io import BytesIO

def predict_single_image(model, img_path, img_size=(150, 150)):
    """
    التنبؤ على صورة واحدة
    """
    # تحميل الصورة (من ملف أو URL)
    if img_path.startswith('http'):
        response = requests.get(img_path)
        img = image.load_img(BytesIO(response.content), target_size=img_size)
    else:
        img = image.load_img(img_path, target_size=img_size)
    
    # تحويل لمصفوفة وتطبيع
    img_array = image.img_to_array(img)
    img_array = np.expand_dims(img_array, axis=0)
    img_array = img_array / 255.0
    
    # التنبؤ
    prediction = model.predict(img_array, verbose=0)[0][0]
    
    # تحديد الفئة
    if prediction > 0.5:
        label = 'Dog'
        confidence = prediction * 100
    else:
        label = 'Cat'
        confidence = (1 - prediction) * 100
    
    # عرض النتيجة
    plt.figure(figsize=(6, 6))
    plt.imshow(img)
    plt.title(f'{label} ({confidence:.1f}%)', fontsize=14, fontweight='bold')
    plt.axis('off')
    plt.show()
    
    return label, confidence

# اختبار على صور متعددة
test_images = [
    'test_images/my_cat.jpg',
    'test_images/my_dog.jpg',
    'https://example.com/random_cat.jpg'
]

for img_path in test_images:
    try:
        label, conf = predict_single_image(model, img_path)
        print(f"{img_path}: {label} ({conf:.1f}%)")
    except Exception as e:
        print(f"خطأ في {img_path}: {e}")
