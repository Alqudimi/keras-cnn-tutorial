def load_dataset_from_directory(data_dir, img_size=(224, 224), 
                                 batch_size=32, validation_split=0.2):
    """
    ุชุญููู ูุฌููุนุฉ ุงูุจูุงูุงุช ูู ุงููุฌูุฏุงุช
    
    Args:
        data_dir: ูุณุงุฑ ูุฌูุฏ ุงูุจูุงูุงุช
        img_size: ุญุฌู ุงูุตูุฑุฉ ุงููุทููุจ
        batch_size: ุญุฌู ุงูุฏูุนุฉ
        validation_split: ูุณุจุฉ ุจูุงูุงุช ุงูุชุญูู
    
    Returns:
        train_ds, val_ds: ูุฌููุนุงุช ุจูุงูุงุช TensorFlow
    """
    
    train_ds = tf.keras.utils.image_dataset_from_directory(
        data_dir,
        validation_split=validation_split,
        subset="training",
        seed=42,
        image_size=img_size,
        batch_size=batch_size,
        label_mode='categorical'
    )
    
    val_ds = tf.keras.utils.image_dataset_from_directory(
        data_dir,
        validation_split=validation_split,
        subset="validation",
        seed=42,
        image_size=img_size,
        batch_size=batch_size,
        label_mode='categorical'
    )
    
    class_names = train_ds.class_names
    print(f"๐ ุงููุฆุงุช ุงูููุชุดูุฉ: {class_names}")
    print(f"๐ ุนุฏุฏ ุฏูุนุงุช ุงูุชุฏุฑูุจ: {len(train_ds)}")
    print(f"๐ ุนุฏุฏ ุฏูุนุงุช ุงูุชุญูู: {len(val_ds)}")
    
    return train_ds, val_ds, class_names


def prepare_dataset(ds, augment=False, shuffle=False, 
                    cache=True, prefetch=True):
    """
    ุฅุนุฏุงุฏ ูุฌููุนุฉ ุงูุจูุงูุงุช ููุชุฏุฑูุจ
    
    Args:
        ds: ูุฌููุนุฉ ุงูุจูุงูุงุช
        augment: ุชุทุจูู ุฒูุงุฏุฉ ุงูุจูุงูุงุช
        shuffle: ุฎูุท ุงูุจูุงูุงุช
        cache: ุชุฎุฒูู ูุคูุช
        prefetch: ุฌูุจ ูุณุจู
    
    Returns:
        ูุฌููุนุฉ ุจูุงูุงุช ูุนุฏุฉ
    """
    AUTOTUNE = tf.data.AUTOTUNE
    
    # ุงูุชุทุจูุน
    normalization = layers.Rescaling(1./255)
    ds = ds.map(lambda x, y: (normalization(x), y))
    
    # ุฒูุงุฏุฉ ุงูุจูุงูุงุช (ููุชุฏุฑูุจ ููุท)
    if augment:
        augmentation = create_augmentation_layer()
        ds = ds.map(lambda x, y: (augmentation(x, training=True), y))
    
    # ุงูุชุฎุฒูู ุงููุคูุช
    if cache:
        ds = ds.cache()
    
    # ุงูุฎูุท
    if shuffle:
        ds = ds.shuffle(1000)
    
    # ุงูุฌูุจ ุงููุณุจู
    if prefetch:
        ds = ds.prefetch(buffer_size=AUTOTUNE)
    
    return ds

def get_callbacks(save_path='models/best_model.h5'):
    """ุฅูุดุงุก Callbacks ููุชุฏุฑูุจ"""
    
    early_stopping = EarlyStopping(
        monitor='val_loss',
        patience=15,
        restore_best_weights=True,
        verbose=1
    )
    
    checkpoint = ModelCheckpoint(
        save_path,
        monitor='val_accuracy',
        save_best_only=True,
        verbose=1
    )
    
    reduce_lr = ReduceLROnPlateau(
        monitor='val_loss',
        factor=0.5,
        patience=5,
        min_lr=1e-7,
        verbose=1
    )
    
    return early_stopping, checkpoint, reduce_lr


def train_three_phases(model, x_train, y_train, x_val, y_val,

def preprocess_image(image_path, target_size=(224, 224)):
    """
    ูุนุงูุฌุฉ ุตูุฑุฉ ุฌุฏูุฏุฉ ููุชูุจุค

    # ูุฑุงุกุฉ ุงูุตูุฑุฉ

    # ุฅุถุงูุฉ ุจูุนุฏ ููุฏูุนุฉ
    img = np.expand_dims(img, axis=0)

def predict_single_image(model, image_path, class_names, 
                         target_size=(224, 224)):
    """
    ุงูุชูุจุค ุจูุฆุฉ ุตูุฑุฉ ูุงุญุฏุฉ

        image_path: ูุณุงุฑ ุงูุตูุฑุฉ
        class_names: ุฃุณูุงุก ุงููุฆุงุช
        target_size: ุงูุญุฌู ุงููุทููุจ
    
    Returns:
        predicted_class: ุงููุฆุฉ ุงููุชููุนุฉ
        confidence: ุฏุฑุฌุฉ ุงูุซูุฉ
        all_probs: ุฌููุน ุงูุงุญุชูุงูุงุช
    """
    
    # ูุนุงูุฌุฉ ุงูุตูุฑุฉ
    img = preprocess_image(image_path, target_size)
    
    # ุงูุชูุจุค
    predictions = model.predict(img, verbose=0)
    
    # ุงุณุชุฎุฑุงุฌ ุงููุชุงุฆุฌ
    predicted_idx = np.argmax(predictions[0])
    predicted_class = class_names[predicted_idx]
    confidence = predictions[0][predicted_idx]
    
    print(f"\n๐ฏ ูุชูุฌุฉ ุงูุชูุจุค:")
    print(f"   - ุงููุฆุฉ: {predicted_class}")

    return predicted_class, confidence, predictions[0]
