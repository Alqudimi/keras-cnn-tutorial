التقييم والمقاييس (Evaluation & Metrics)

دليل شامل لفهم واستخدام مقاييس تقييم نماذج التعلم العميق. تعلم كيفية قياس أداء نموذجك وتحليل الأخطاء وتحسين النتائج.

مقدمة: لماذا التقييم مهم؟

التقييم الصحيح للنموذج هو المفتاح لفهم أدائه الحقيقي وتحديد نقاط الضعف التي تحتاج للتحسين. لا يكفي أن نعرف أن النموذج "يعمل"، بل نحتاج لفهم كيف يعمل ومتى يفشل.

نقطة مهمة

Accuracy وحده ليس كافياً! في البيانات غير المتوازنة، قد يحقق النموذج 95% accuracy بمجرد تصنيف كل شيء كفئة واحدة. لذلك نحتاج لمقاييس متعددة.

Accuracy (الدقة الإجمالية)

Accuracy هو أبسط المقاييس: نسبة التنبؤات الصحيحة من إجمالي التنبؤات.

Formula 
Accuracy = (عدد التنبؤات الصحيحة) / (إجمالي التنبؤات)

         = (TP + TN) / (TP + TN + FP + FN)

حيث:
- TP (True Positive): توقع صحيح للفئة الإيجابية
- TN (True Negative): توقع صحيح للفئة السلبية
- FP (False Positive): توقع خاطئ للفئة الإيجابية
- FN (False Negative): توقع خاطئ للفئة السلبية
Python 
from sklearn.metrics import accuracy_score
import numpy as np

# حساب الدقة
y_true = [0, 1, 1, 0, 1, 0, 1, 1]
y_pred = [0, 1, 0, 0, 1, 1, 1, 1]

accuracy = accuracy_score(y_true, y_pred)
print(f"Accuracy: {accuracy:.2%}")  # 75.00%

# باستخدام TensorFlow/Keras
loss, accuracy = model.evaluate(test_dataset)
print(f"Test Accuracy: {accuracy:.2%}")

متى لا تستخدم Accuracy فقط؟

إذا كان لديك 950 صورة قطط و 50 صورة كلاب، النموذج الذي يتوقع "قط" دائماً سيحقق 95% accuracy! لذلك في البيانات غير المتوازنة، استخدم مقاييس أخرى.

Precision و Recall

Precision (الدقة)

من بين كل ما توقعه النموذج كفئة معينة، كم منها كان صحيحاً فعلاً؟

Recall (الاسترجاع)

من بين كل العينات الفعلية لفئة معينة، كم منها اكتشفه النموذج؟

Formula 
Precision = TP / (TP + FP)
مثال: إذا توقع النموذج 100 صورة كـ "قط"، و80 منها فعلاً قطط
Precision = 80/100 = 80%

Recall = TP / (TP + FN)  
مثال: إذا كان هناك 100 صورة قط فعلية، والنموذج اكتشف 80 منها
Recall = 80/100 = 80%
Python 
from sklearn.metrics import precision_score, recall_score, classification_report

# للتصنيف الثنائي
precision = precision_score(y_true, y_pred)
recall = recall_score(y_true, y_pred)

print(f"Precision: {precision:.2%}")
print(f"Recall: {recall:.2%}")

# للتصنيف المتعدد
precision = precision_score(y_true, y_pred, average='weighted')
recall = recall_score(y_true, y_pred, average='weighted')

# تقرير شامل لكل فئة
print(classification_report(y_true, y_pred, target_names=class_names))

متى نفضل أيهما؟

الحالة المقياس المفضل السبب
كشف السبام Precision عالي لا نريد حذف رسائل مهمة بالخطأ
كشف السرطان Recall عالي لا نريد أن يفوتنا مريض حقيقي
نظام أمني Recall عالي لا نريد أن تفوتنا تهديدات حقيقية
توصيات المنتجات Precision عالي توصيات خاطئة تضر بتجربة المستخدم

F1-Score

F1-Score هو المتوسط التوافقي بين Precision و Recall. يعطي توازناً بين المقياسين ويكون مفيداً عندما تريد مقياساً واحداً يجمع بينهما.

Formula 
F1 = 2 × (Precision × Recall) / (Precision + Recall)

مثال:
Precision = 80%
Recall = 60%
F1 = 2 × (0.80 × 0.60) / (0.80 + 0.60) = 0.685 = 68.5%
Python 
from sklearn.metrics import f1_score

# للتصنيف الثنائي
f1 = f1_score(y_true, y_pred)

# للتصنيف المتعدد - عدة طرق للحساب
f1_micro = f1_score(y_true, y_pred, average='micro')    # إجمالي
f1_macro = f1_score(y_true, y_pred, average='macro')    # متوسط الفئات
f1_weighted = f1_score(y_true, y_pred, average='weighted')  # موزون

print(f"F1 (micro): {f1_micro:.2%}")
print(f"F1 (macro): {f1_macro:.2%}")
print(f"F1 (weighted): {f1_weighted:.2%}")

# F1 لكل فئة
f1_per_class = f1_score(y_true, y_pred, average=None)
for i, f1 in enumerate(f1_per_class):
    print(f"Class {i}: F1 = {f1:.2%}")

مصفوفة الارتباك (Confusion Matrix)

مصفوفة الارتباك تعطي صورة كاملة عن أداء النموذج: أين يصيب وأين يخطئ، وما هي الفئات التي يخلط بينها.

Python 
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.metrics import confusion_matrix

def plot_confusion_matrix(y_true, y_pred, class_names, figsize=(10, 8)):
    """
    رسم مصفوفة الارتباك
    """
    cm = confusion_matrix(y_true, y_pred)
    
    # تحويل إلى نسب مئوية
    cm_percent = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis] * 100
    
    fig, axes = plt.subplots(1, 2, figsize=figsize)
    
    # المصفوفة بالأعداد
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
                xticklabels=class_names, yticklabels=class_names,
                ax=axes[0])
    axes[0].set_title('Confusion Matrix (Counts)')
    axes[0].set_xlabel('Predicted')
    axes[0].set_ylabel('Actual')
    
    # المصفوفة بالنسب المئوية
    sns.heatmap(cm_percent, annot=True, fmt='.1f', cmap='Blues',
                xticklabels=class_names, yticklabels=class_names,
                ax=axes[1])
    axes[1].set_title('Confusion Matrix (Percentages)')
    axes[1].set_xlabel('Predicted')
    axes[1].set_ylabel('Actual')
    
    plt.tight_layout()
    plt.savefig('confusion_matrix.png', dpi=150)
    plt.show()
    
    return cm

# الاستخدام مع نموذج Keras

ROC و AUC

منحنى ROC (Receiver Operating Characteristic) يوضح العلاقة بين True Positive Rate و False Positive Rate عند عتبات (thresholds) مختلفة. AUC هي المساحة تحت هذا المنحنى.

Python 
from sklearn.metrics import roc_curve, auc, roc_auc_score
from sklearn.preprocessing import label_binarize
import matplotlib.pyplot as plt

def plot_roc_curves(y_true, y_pred_probs, class_names):
    """
    رسم منحنيات ROC لجميع الفئات
    """
    n_classes = len(class_names)
    
    # تحويل التسميات إلى one-hot
    y_true_bin = label_binarize(y_true, classes=range(n_classes))
    
    plt.figure(figsize=(10, 8))
    
    colors = plt.cm.get_cmap('tab10')(np.linspace(0, 1, n_classes))
    
    for i, (class_name, color) in enumerate(zip(class_names, colors)):
        fpr, tpr, _ = roc_curve(y_true_bin[:, i], y_pred_probs[:, i])
        roc_auc = auc(fpr, tpr)
        
        plt.plot(fpr, tpr, color=color, lw=2,
                 label=f'{class_name} (AUC = {roc_auc:.2f})')
    
    # الخط القطري (التخمين العشوائي)
    plt.plot([0, 1], [0, 1], 'k--', lw=2, label='Random (AUC = 0.50)')
    
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('ROC Curves for Multi-class Classification')
    plt.legend(loc='lower right')
    plt.grid(True, alpha=0.3)
    plt.savefig('roc_curves.png', dpi=150)
    plt.show()

# حساب AUC الإجمالي
auc_score = roc_auc_score(y_true, y_pred_probs, multi_class='ovr')
print(f"Overall AUC: {auc_score:.2f}")

# الاستخدام
y_pred_probs = model.predict(test_dataset)
plot_roc_curves(y_true, y_pred_probs, class_names)

تحليل الأخطاء (Error Analysis)

تحليل الأخطاء هو عملية فحص الحالات التي أخطأ فيها النموذج لفهم أنماط الخطأ وتحديد كيفية تحسينه.

Python 
import matplotlib.pyplot as plt
import numpy as np

def analyze_errors(model, test_dataset, class_names, num_samples=20):
    """
    تحليل وعرض الأخطاء
    """
    errors = []
    
    for images, labels in test_dataset:
        predictions = model.predict(images, verbose=0)
        pred_classes = np.argmax(predictions, axis=1)
        
        for i in range(len(images)):
            true_class = labels[i].numpy() if hasattr(labels[i], 'numpy') else labels[i]
            pred_class = pred_classes[i]
            confidence = predictions[i][pred_class]
            
            if true_class != pred_class:
                errors.append({
                    'image': images[i].numpy(),
                    'true': true_class,
                    'predicted': pred_class,
                    'confidence': confidence,
                    'probabilities': predictions[i]
                })
    
    print(f"Total errors: {len(errors)}")
    
    # عرض بعض الأخطاء
    fig, axes = plt.subplots(4, 5, figsize=(15, 12))
    
    for idx, ax in enumerate(axes.flat):
        if idx >= len(errors) or idx >= num_samples:
            ax.axis('off')
            continue
            
        error = errors[idx]
        img = error['image']
        
        # تطبيع الصورة للعرض
        if img.max() <= 1:
            img = (img * 255).astype(np.uint8)
        
        ax.imshow(img)
        ax.set_title(f"True: {class_names[error['true']]}\n"
                     f"Pred: {class_names[error['predicted']]} "
                     f"({error['confidence']:.1%})",
                     fontsize=9)
        ax.axis('off')
    
    plt.tight_layout()
    plt.savefig('error_analysis.png', dpi=150)
    plt.show()
    
    return errors

# تحليل أنماط الأخطاء
def error_patterns(errors, class_names):
    """
    تحليل أنماط الأخطاء
    """
    error_counts = {}
    
    for error in errors:
        key = (class_names[error['true']], class_names[error['predicted']])
        error_counts[key] = error_counts.get(key, 0) + 1
    
    # ترتيب حسب التكرار
    sorted_errors = sorted(error_counts.items(), key=lambda x: x[1], reverse=True)
    
    print("\nMost Common Errors:")
    print("-" * 50)
    for (true, pred), count in sorted_errors[:10]:
        print(f"{true} -> {pred}: {count} times")
    
    return sorted_errors

# الاستخدام
errors = analyze_errors(model, test_dataset, class_names)
patterns = error_patterns(errors, class_names)

def high_confidence_errors(errors, threshold=0.9):
    """
    الأخطاء التي كان النموذج واثقاً منها جداً
    هذه هي الأخطاء الأكثر خطورة!
    """
    high_conf = [e for e in errors if e['confidence'] >= threshold]
    
    print(f"\nHigh confidence errors (>= {threshold:.0%}): {len(high_conf)}")
    
    for error in high_conf[:5]:
        print(f"  True: {class_names[error['true']]}, "
              f"Predicted: {class_names[error['predicted']]} "
              f"with {error['confidence']:.1%} confidence")
    
    return high_conf

1. جمع المزيد من البيانات

Python 
def identify_weak_classes(y_true, y_pred, class_names, threshold=0.7):
    """
    تحديد الفئات التي تحتاج المزيد من البيانات
    """
    from sklearn.metrics import precision_recall_fscore_support
    
    precision, recall, f1, support = precision_recall_fscore_support(
        y_true, y_pred, average=None
    )
    
    weak_classes = []
    
    print("\nClass Performance Analysis:")
    print("-" * 60)
    
    for i, class_name in enumerate(class_names):
        status = "WEAK" if f1[i] < threshold else "OK"
        print(f"{class_name:15} | F1: {f1[i]:.2%} | "
              f"P: {precision[i]:.2%} | R: {recall[i]:.2%} | "
              f"N: {support[i]:4} | {status}")
        
        if f1[i] < threshold:
            weak_classes.append({
                'class': class_name,
                'f1': f1[i],
                'precision': precision[i],
                'recall': recall[i],
                'samples': support[i]
            })
    
    return weak_classes

# استخدام أوزان مختلفة للفئات
def calculate_focal_weights(class_f1_scores, gamma=2):
    """
    حساب أوزان تركز على الفئات الصعبة
    """
    weights = {}
    for i, f1 in enumerate(class_f1_scores):
        # الفئات ذات F1 المنخفض تحصل على وزن أعلى
        weights[i] = (1 - f1) ** gamma
    
    # تطبيع
    total = sum(weights.values())
    weights = {k: v / total * len(weights) for k, v in weights.items()}
    
    return weights

def create_balanced_dataset(dataset, weak_classes, oversample_factor=3):
    """
    إنشاء dataset متوازن مع تكرار الفئات الضعيفة
    """
    weak_class_indices = [c['class_idx'] for c in weak_classes]
    
    balanced_images = []
    balanced_labels = []
    
    for images, labels in dataset:
        for i in range(len(images)):
            label = labels[i].numpy()
            
            # إضافة الصورة
            balanced_images.append(images[i])
            balanced_labels.append(label)
            
            # تكرار الفئات الضعيفة
            if label in weak_class_indices:
                for _ in range(oversample_factor - 1):
                    balanced_images.append(images[i])
                    balanced_labels.append(label)
    
    return tf.data.Dataset.from_tensor_slices(
        (np.array(balanced_images), np.array(balanced_labels))
    ).shuffle(1000).batch(32)
