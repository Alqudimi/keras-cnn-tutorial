    ุญุณุงุจ ุฃูุฒุงู ุงููุฆุงุช ููุชุนุงูู ูุน ุนุฏู ุงูุชูุงุฒู

from tensorflow.keras import Model
from tensorflow.keras.layers import (
    Input, Conv2D, BatchNormalization, Activation, MaxPooling2D,
    GlobalAveragePooling2D, Dense, Dropout, Add, Multiply, Reshape
)

def squeeze_excite_block(x, ratio=16):
    """
    Squeeze-and-Excitation block ูุชุญุณูู ุชูุซูู ุงูููุฒุงุช
    """
    filters = x.shape[-1]
    se = GlobalAveragePooling2D()(x)
    se = Reshape((1, 1, filters))(se)
    se = Dense(filters // ratio, activation='relu')(se)
    se = Dense(filters, activation='sigmoid')(se)
    return Multiply()([x, se])

def conv_block(x, filters, kernel_size=3, use_se=True):
    """
    ูุชูุฉ ุชูุงููููุฉ ูุน BatchNorm ู SE
    """
    x = Conv2D(filters, kernel_size, padding='same')(x)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    x = Conv2D(filters, kernel_size, padding='same')(x)
    x = BatchNormalization()(x)
    
    if use_se:
        x = squeeze_excite_block(x)
    
    x = Activation('relu')(x)
    return x

def build_flowers_model(input_shape=(224, 224, 3), num_classes=5):
    """
    ุจูุงุก ูููุฐุฌ CNN ุนููู ูุชุตููู ุงูุฒููุฑ
    """
    inputs = Input(shape=input_shape)
    
    # ุงููุชูุฉ ุงูุฃููู
    x = conv_block(inputs, 32, use_se=False)
    x = MaxPooling2D((2, 2))(x)
    x = Dropout(0.25)(x)
    
    # ุงููุชูุฉ ุงูุซุงููุฉ
    x = conv_block(x, 64)
    x = MaxPooling2D((2, 2))(x)
    x = Dropout(0.25)(x)
    
    # ุงููุชูุฉ ุงูุซุงูุซุฉ
    x = conv_block(x, 128)
    x = MaxPooling2D((2, 2))(x)
    x = Dropout(0.3)(x)
    
    # ุงููุชูุฉ ุงูุฑุงุจุนุฉ
    x = conv_block(x, 256)
    x = MaxPooling2D((2, 2))(x)
    x = Dropout(0.3)(x)
    
    # ุงููุชูุฉ ุงูุฎุงูุณุฉ
    x = conv_block(x, 512)
    x = MaxPooling2D((2, 2))(x)
    x = Dropout(0.4)(x)
    
    # ุทุจูุงุช ุงูุชุตููู
    x = GlobalAveragePooling2D()(x)
    x = Dense(512, activation='relu')(x)
    x = Dropout(0.5)(x)
    x = Dense(256, activation='relu')(x)
    x = Dropout(0.4)(x)
    outputs = Dense(num_classes, activation='softmax')(x)
    
    model = Model(inputs, outputs, name='flowers_classifier')
    return model

def show_misclassified_images(model, test_ds, class_names, num_images=16):
    """
    ุนุฑุถ ุงูุตูุฑ ุงูุชู ุชู ุชุตููููุง ุจุดูู ุฎุงุทุฆ
    """
    misclassified = []
    
    for images, labels in test_ds:
        predictions = model.predict(images, verbose=0)
        pred_classes = np.argmax(predictions, axis=1)
        true_classes = np.argmax(labels.numpy(), axis=1)
        
        for i in range(len(images)):
            if pred_classes[i] != true_classes[i]:
                misclassified.append({
                    'image': images[i].numpy(),
                    'true': class_names[true_classes[i]],
                    'pred': class_names[pred_classes[i]],
                    'confidence': predictions[i][pred_classes[i]] * 100
                })
        
        if len(misclassified) >= num_images:
            break
    
    # ุนุฑุถ ุงูุตูุฑ
    cols = 4
    rows = (num_images + cols - 1) // cols
    fig, axes = plt.subplots(rows, cols, figsize=(16, 4*rows))
    axes = axes.flatten()
    
    for idx, item in enumerate(misclassified[:num_images]):
        axes[idx].imshow(item['image'])
        axes[idx].set_title(
            f"ุญูููู: {item['true']}\nูุชููุน: {item['pred']} ({item['confidence']:.1f}%)",
            fontsize=10, color='red'
        )
        axes[idx].axis('off')
    
    # ุฅุฎูุงุก ุงููุญุงูุฑ ุงููุงุฑุบุฉ
    for idx in range(len(misclassified), len(axes)):
        axes[idx].axis('off')
    
    plt.suptitle('ุงูุตูุฑ ุงููุตููุฉ ุจุดูู ุฎุงุทุฆ', fontsize=14, fontweight='bold')
    plt.tight_layout()
    plt.savefig('misclassified_flowers.png', dpi=150)
    plt.show()
    
    # ุฅุญุตุงุฆูุงุช ุงูุฃุฎุทุงุก
    print("\nุฅุญุตุงุฆูุงุช ุงูุฃุฎุทุงุก ุงูุดุงุฆุนุฉ:")
