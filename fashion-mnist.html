مشروع تصنيف الملابس (Fashion-MNIST)

تعلم بناء CNN للصور الرمادية باستخدام مجموعة Fashion-MNIST الشهيرة. 10 فئات من الملابس بصور بحجم 28×28 بكسل - مثالي للتدريب السريع والفعال.

مقدمة عن Fashion-MNIST

Fashion-MNIST هي مجموعة بيانات طورتها شركة Zalando كبديل عصري لـ MNIST الكلاسيكية. تحتوي على 70,000 صورة رمادية بحجم 28×28 بكسل لـ 10 أنواع مختلفة من الملابس والإكسسوارات.

مواصفات البيانات

  • حجم الصورة: 28 × 28 بكسل (رمادية)
  • التدريب: 60,000 صورة
  • الاختبار: 10,000 صورة
  • الفئات: 10 أنواع ملابس

تدريب سريع

صور صغيرة (28×28) تسمح بتدريب سريع حتى على CPU

تحدي حقيقي

لماذا استبدال MNIST؟

MNIST أصبحت سهلة جدًا (99%+ accuracy بنماذج بسيطة). Fashion-MNIST أصعب وأكثر تمثيلاً للتحديات الحقيقية مع الحفاظ على نفس التنسيق.

تحميل البيانات من Keras

import tensorflow as tf
from tensorflow.keras.datasets import fashion_mnist
import numpy as np
import matplotlib.pyplot as plt

# تحميل البيانات مباشرة من Keras
(X_train, y_train), (X_test, y_test) = fashion_mnist.load_data()

# أسماء الفئات
class_names = [
    'T-shirt/top', 'Trouser', 'Pullover', 'Dress', 'Coat',
    'Sandal', 'Shirt', 'Sneaker', 'Bag', 'Ankle boot'
]

# معلومات عن البيانات
print("معلومات البيانات:")
print(f"شكل بيانات التدريب: {X_train.shape}")
print(f"شكل بيانات الاختبار: {X_test.shape}")
print(f"نوع البيانات: {X_train.dtype}")
print(f"نطاق القيم: {X_train.min()} - {X_train.max()}")

# عرض بعض الصور
fig, axes = plt.subplots(2, 5, figsize=(12, 5))
for i, ax in enumerate(axes.flatten()):
    ax.imshow(X_train[i], cmap='gray')
    ax.set_title(class_names[y_train[i]])
    ax.axis('off')
plt.suptitle('عينات من البيانات', fontsize=14, fontweight='bold')
plt.tight_layout()
plt.show()

تطبيع الصور وتغيير الشكل

الصور الرمادية لها قناة واحدة فقط، لذا يجب إضافة بُعد إضافي لتتوافق مع توقعات CNN. كما يجب تطبيع القيم من [0, 255] إلى [0, 1].

Python 
# تطبيع الصور (من 0-255 إلى 0-1)
X_train = X_train.astype('float32') / 255.0
X_test = X_test.astype('float32') / 255.0

# تغيير الشكل لإضافة بُعد القناة
# من (60000, 28, 28) إلى (60000, 28, 28, 1)
X_train = X_train.reshape(-1, 28, 28, 1)
X_test = X_test.reshape(-1, 28, 28, 1)

print(f"شكل البيانات بعد التحويل:")
print(f"التدريب: {X_train.shape}")
print(f"الاختبار: {X_test.shape}")

# تحويل التسميات إلى one-hot encoding
y_train_cat = tf.keras.utils.to_categorical(y_train, 10)
y_test_cat = tf.keras.utils.to_categorical(y_test, 10)

print(f"\nشكل التسميات بعد One-Hot:")
print(f"التدريب: {y_train_cat.shape}")
print(f"الاختبار: {y_test_cat.shape}")

# تقسيم بيانات التدريب إلى تدريب وتحقق
from sklearn.model_selection import train_test_split

X_train, X_val, y_train_cat, y_val_cat = train_test_split(
    X_train, y_train_cat, test_size=0.1, random_state=42
)

print(f"\nبعد التقسيم:")
print(f"التدريب: {X_train.shape}")
print(f"التحقق: {X_val.shape}")

ملاحظة مهمة

شكل الإدخال يجب أن يكون (28, 28, 1) وليس (28, 28). بُعد القناة ضروري حتى للصور الرمادية عند استخدام طبقات Conv2D.

بناء CNN مناسبة للصور الرمادية

سنبني شبكة خفيفة ومناسبة لحجم الصور الصغير. لا نحتاج لشبكة عميقة جدًا لأن الصور بسيطة نسبيًا.

Python 
from tensorflow.keras import Sequential
from tensorflow.keras.layers import (
    Conv2D, BatchNormalization, Activation, MaxPooling2D,
    GlobalAveragePooling2D, Dense, Dropout
)

def build_fashion_model(input_shape=(28, 28, 1), num_classes=10):
    """
    بناء نموذج CNN خفيف لـ Fashion-MNIST
    """
    model = Sequential([
        # الكتلة الأولى
        Conv2D(32, (3, 3), padding='same', input_shape=input_shape),
        BatchNormalization(),
        Activation('relu'),
        Conv2D(32, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        MaxPooling2D((2, 2)),
        Dropout(0.25),
        
        # الكتلة الثانية
        Conv2D(64, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        Conv2D(64, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        MaxPooling2D((2, 2)),
        Dropout(0.25),
        
        # الكتلة الثالثة
        Conv2D(128, (3, 3), padding='same'),
        BatchNormalization(),
        Activation('relu'),
        Dropout(0.4),
        
        # طبقات التصنيف
        GlobalAveragePooling2D(),
        Dense(256, activation='relu'),
        BatchNormalization(),
        Dropout(0.5),
        Dense(num_classes, activation='softmax')
    ])
    
    return model

# إنشاء النموذج
model = build_fashion_model()
model.summary()

# عرض ملخص مختصر
print(f"\nإجمالي المعاملات: {model.count_params():,}")
print(f"المعاملات القابلة للتدريب: {sum([tf.keras.backend.count_params(w) for w in model.trainable_weights]):,}")

التدريب السريع والدقيق

Python 
from tensorflow.keras.optimizers import Adam
from tensorflow.keras.callbacks import (
    EarlyStopping, ModelCheckpoint, ReduceLROnPlateau
)

# تجميع النموذج
model.compile(
    optimizer=Adam(learning_rate=0.001),
    loss='categorical_crossentropy',
    metrics=['accuracy']
)

# Callbacks
callbacks = [
    EarlyStopping(
        monitor='val_loss',
        patience=10,
        restore_best_weights=True,
        verbose=1
    ),
    ModelCheckpoint(
        'best_fashion_model.keras',
        monitor='val_accuracy',
        save_best_only=True,
        verbose=1
    ),
    ReduceLROnPlateau(
        monitor='val_loss',
        factor=0.5,
        patience=5,
        min_lr=1e-6,
        verbose=1
    )
]

# التدريب
history = model.fit(
    X_train, y_train_cat,
    batch_size=128,
    epochs=50,
    validation_data=(X_val, y_val_cat),
    callbacks=callbacks,
    verbose=1
)

# رسم منحنيات التدريب
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

axes[0].plot(history.history['accuracy'], label='Training')
axes[0].plot(history.history['val_accuracy'], label='Validation')
axes[0].set_title('Model Accuracy')
axes[0].set_xlabel('Epoch')
axes[0].set_ylabel('Accuracy')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

axes[1].plot(history.history['loss'], label='Training')
axes[1].plot(history.history['val_loss'], label='Validation')
axes[1].set_title('Model Loss')
axes[1].set_xlabel('Epoch')
axes[1].set_ylabel('Loss')
axes[1].legend()
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('fashion_training_curves.png', dpi=150)
plt.show()

مقارنة التوقعات بالحقيقة

Python 
def show_predictions_grid(X, y_true, y_pred, class_names, n_samples=25):
    """
    عرض شبكة من التوقعات مع تمييز الصحيحة والخاطئة
    """
    # اختيار عينات عشوائية
    indices = np.random.choice(len(X), n_samples, replace=False)
    
    cols = 5
    rows = (n_samples + cols - 1) // cols
    fig, axes = plt.subplots(rows, cols, figsize=(15, 3*rows))
    axes = axes.flatten()
    
    for i, idx in enumerate(indices):
        ax = axes[i]
        ax.imshow(X[idx].reshape(28, 28), cmap='gray')
        
        true_label = class_names[y_true[idx]]
        pred_label = class_names[y_pred[idx]]
        
        if y_true[idx] == y_pred[idx]:
            color = 'green'
            title = f'{pred_label}'
        else:
            color = 'red'
            title = f'T: {true_label}\nP: {pred_label}'
        
        ax.set_title(title, fontsize=9, color=color)
        ax.axis('off')
    
    plt.suptitle('مقارنة التوقعات (أخضر=صحيح، أحمر=خطأ)', 
                 fontsize=14, fontweight='bold')
    plt.tight_layout()
    plt.savefig('fashion_predictions_grid.png', dpi=150)
    plt.show()

# عرض الشبكة
show_predictions_grid(X_test, y_test, y_pred_classes, class_names)

# إحصائيات الدقة لكل فئة
print("\nدقة كل فئة:")
for i, name in enumerate(class_names):
    mask = y_test == i
    acc = (y_pred_classes[mask] == y_test[mask]).mean()
    print(f"  {name:15s}: {acc*100:.1f}%")

اختبار النموذج على عينات جديدة

Python 
def predict_custom_image(model, img_path, class_names):
    """
    التنبؤ على صورة مخصصة (يجب تحويلها للتدرج الرمادي)
    """
    from PIL import Image
    
    # تحميل وتحويل الصورة
    img = Image.open(img_path).convert('L')  # تحويل للرمادي
    img = img.resize((28, 28))
    
    # تحويل لمصفوفة
    img_array = np.array(img).astype('float32') / 255.0
    
    # عكس الألوان إذا كانت الخلفية بيضاء
    if img_array.mean() > 0.5:
        img_array = 1 - img_array
    
    img_array = img_array.reshape(1, 28, 28, 1)
    
    # التنبؤ
    predictions = model.predict(img_array, verbose=0)[0]
    top_3 = np.argsort(predictions)[-3:][::-1]
    
    # عرض النتائج
    fig, axes = plt.subplots(1, 2, figsize=(10, 4))
    
    axes[0].imshow(img_array.reshape(28, 28), cmap='gray')
    axes[0].set_title(f'التوقع: {class_names[top_3[0]]}', fontsize=12)
    axes[0].axis('off')
    
    # Top 3 predictions
    colors = ['green', 'steelblue', 'steelblue']
    bars = axes[1].barh([class_names[i] for i in top_3], 
                        [predictions[i]*100 for i in top_3], 
                        color=colors)
    axes[1].set_xlabel('الاحتمالية (%)')
    axes[1].set_title('أعلى 3 توقعات')
    axes[1].set_xlim(0, 100)
    
    plt.tight_layout()
    plt.show()
    
    return class_names[top_3[0]], predictions[top_3[0]]

# اختبار على صورة عشوائية من بيانات الاختبار
