"""
قالب تدريب CNN كامل
"""
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers
import matplotlib.pyplot as plt
import numpy as np
import os

# ================================
# الإعدادات
# ================================
CONFIG = {
    'data_dir': 'dataset',
    'img_size': (224, 224),
    'batch_size': 32,
    'epochs': 50,
    'num_classes': 10,
    'learning_rate': 1e-3,
    'model_save_path': 'best_model.keras'
}

# ================================
# تحميل البيانات
# ================================
def load_data():
    train_ds = keras.utils.image_dataset_from_directory(
        f"{CONFIG['data_dir']}/train",
        image_size=CONFIG['img_size'],
        batch_size=CONFIG['batch_size'],
        label_mode='int'
    )
    
    val_ds = keras.utils.image_dataset_from_directory(
        f"{CONFIG['data_dir']}/validation",
        image_size=CONFIG['img_size'],
        batch_size=CONFIG['batch_size'],
        label_mode='int'
    )
    
    class_names = train_ds.class_names
    print(f"Classes: {class_names}")
    
    AUTOTUNE = tf.data.AUTOTUNE
    train_ds = train_ds.cache().shuffle(1000).prefetch(AUTOTUNE)
    val_ds = val_ds.cache().prefetch(AUTOTUNE)
    
    return train_ds, val_ds, class_names

# ================================
# بناء النموذج
# ================================
def build_model():
    data_augmentation = keras.Sequential([
        layers.RandomFlip("horizontal"),
        layers.RandomRotation(0.1),
        layers.RandomZoom(0.1),
    ])
    
    model = keras.Sequential([
        layers.Rescaling(1./255, input_shape=(*CONFIG['img_size'], 3)),
        data_augmentation,
        
        layers.Conv2D(32, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.MaxPooling2D(),
        
        layers.Conv2D(64, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.MaxPooling2D(),
        
        layers.Conv2D(128, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.MaxPooling2D(),
        
        layers.Conv2D(256, 3, padding='same'),
        layers.BatchNormalization(),
        layers.ReLU(),
        layers.MaxPooling2D(),
        
        layers.GlobalAveragePooling2D(),
        layers.Dropout(0.5),
        layers.Dense(256, activation='relu'),
        layers.Dropout(0.3),
        layers.Dense(CONFIG['num_classes'], activation='softmax')
    ])
    
    return model

# ================================
# التدريب
# ================================
def train():
    train_ds, val_ds, class_names = load_data()
    model = build_model()
    
    model.compile(
        optimizer=keras.optimizers.Adam(CONFIG['learning_rate']),
        loss='sparse_categorical_crossentropy',
        metrics=['accuracy']
    )
    
    model.summary()
    
    callbacks = [
        keras.callbacks.EarlyStopping(
            monitor='val_loss',
            patience=10,
            restore_best_weights=True
        ),
        keras.callbacks.ReduceLROnPlateau(
            monitor='val_loss',
            factor=0.5,
            patience=5,
            min_lr=1e-7
        ),
        keras.callbacks.ModelCheckpoint(
            CONFIG['model_save_path'],
            monitor='val_accuracy',
            save_best_only=True
        )
    ]
    
    history = model.fit(
        train_ds,
        validation_data=val_ds,
        epochs=CONFIG['epochs'],
        callbacks=callbacks
    )
    
    return model, history, class_names

# ================================
# رسم النتائج
# ================================
def plot_history(history):
    fig, axes = plt.subplots(1, 2, figsize=(14, 5))
    
    axes[0].plot(history.history['accuracy'], label='Train')
    axes[0].plot(history.history['val_accuracy'], label='Validation')
    axes[0].set_title('Accuracy')
    axes[0].set_xlabel('Epoch')
    axes[0].legend()
    
    axes[1].plot(history.history['loss'], label='Train')
    axes[1].plot(history.history['val_loss'], label='Validation')
    axes[1].set_title('Loss')
    axes[1].set_xlabel('Epoch')
    axes[1].legend()
    
    plt.tight_layout()
    plt.savefig('training_history.png', dpi=150)
    plt.show()

# ================================
# التشغيل
# ================================
if __name__ == "__main__":
    model, history, class_names = train()
    plot_history(history)
    
    print(f"\nBest validation accuracy: {max(history.history['val_accuracy']):.2%}")
    print(f"Model saved to: {CONFIG['model_save_path']}")

"""
قالب Data Augmentation شامل
"""
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers
import matplotlib.pyplot as plt
import numpy as np

# ================================
# Augmentation خفيف (للبداية)
# ================================
light_augmentation = keras.Sequential([
    layers.RandomFlip("horizontal"),
    layers.RandomRotation(0.05),
], name="light_augmentation")

# ================================
# Augmentation متوسط (موصى به)
# ================================
medium_augmentation = keras.Sequential([
    layers.RandomFlip("horizontal"),
    layers.RandomRotation(0.1),
    layers.RandomZoom(0.1),
    layers.RandomContrast(0.1),
], name="medium_augmentation")

# ================================
# Augmentation قوي (للبيانات القليلة)
# ================================
strong_augmentation = keras.Sequential([
    layers.RandomFlip("horizontal_and_vertical"),
    layers.RandomRotation(0.2),
    layers.RandomZoom(0.2),
    layers.RandomContrast(0.2),
    layers.RandomTranslation(0.1, 0.1),
], name="strong_augmentation")

# ================================
# Augmentation مخصص
# ================================
class CustomAugmentation(layers.Layer):
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        self.flip = layers.RandomFlip("horizontal")
        self.rotation = layers.RandomRotation(0.1)
        self.zoom = layers.RandomZoom(0.1)
        
    def call(self, inputs, training=None):
        if training:
            x = self.flip(inputs)
            x = self.rotation(x)
            x = self.zoom(x)
            return x
        return inputs

# ================================
# تصور Augmentation
# ================================
def visualize_augmentation(image_path, augmentation, n_samples=9):
    img = keras.utils.load_img(image_path, target_size=(224, 224))
    img_array = keras.utils.img_to_array(img)
    img_array = tf.expand_dims(img_array, 0)
    
    fig, axes = plt.subplots(3, 3, figsize=(10, 10))
    
    for ax in axes.flat:
        augmented = augmentation(img_array)
        ax.imshow(augmented[0].numpy().astype("uint8"))
        ax.axis('off')
    
    plt.suptitle(f'Augmentation: {augmentation.name}')
    plt.tight_layout()
    plt.savefig(f'{augmentation.name}_samples.png', dpi=150)
    plt.show()

# ================================
# استخدام في Pipeline
# ================================
def create_augmented_dataset(data_dir, augmentation, batch_size=32):
    dataset = keras.utils.image_dataset_from_directory(
        data_dir,
        image_size=(224, 224),
        batch_size=batch_size
    )
    
    dataset = dataset.map(
        lambda x, y: (augmentation(x, training=True), y),
        num_parallel_calls=tf.data.AUTOTUNE
    )
    
    return dataset.prefetch(tf.data.AUTOTUNE)

# ================================
# الاستخدام
# ================================
if __name__ == "__main__":
    visualize_augmentation("sample_image.jpg", medium_augmentation)

"""
قالب تقييم شامل للنموذج
"""
import tensorflow as tf
from tensorflow import keras
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.metrics import (
    classification_report,
    confusion_matrix,
    precision_recall_fscore_support,
    roc_curve,
    auc
)

def evaluate_model(model, test_dataset, class_names, output_dir='evaluation'):
    """
    تقييم شامل للنموذج
    """
    import os
    os.makedirs(output_dir, exist_ok=True)
    
    y_true = []
    y_pred = []
    y_pred_proba = []
    
    for images, labels in test_dataset:
        predictions = model.predict(images, verbose=0)
        y_true.extend(labels.numpy())
        y_pred.extend(np.argmax(predictions, axis=1))
        y_pred_proba.extend(predictions)
    
    y_true = np.array(y_true)
    y_pred = np.array(y_pred)
    y_pred_proba = np.array(y_pred_proba)
    
    print("=" * 60)
    print("تقرير التصنيف")
    print("=" * 60)
    print(classification_report(y_true, y_pred, target_names=class_names))
    
    plot_confusion_matrix(y_true, y_pred, class_names, output_dir)
    plot_roc_curves(y_true, y_pred_proba, class_names, output_dir)
    analyze_errors(y_true, y_pred, class_names)
    
    return {
        'y_true': y_true,
        'y_pred': y_pred,
        'y_pred_proba': y_pred_proba
    }

def plot_confusion_matrix(y_true, y_pred, class_names, output_dir):
    cm = confusion_matrix(y_true, y_pred)
    
    plt.figure(figsize=(10, 8))
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
                xticklabels=class_names, yticklabels=class_names)
    plt.title('Confusion Matrix')
    plt.xlabel('Predicted')
    plt.ylabel('Actual')
    plt.tight_layout()
    plt.savefig(f'{output_dir}/confusion_matrix.png', dpi=150)
    plt.show()

def plot_roc_curves(y_true, y_pred_proba, class_names, output_dir):
    from sklearn.preprocessing import label_binarize
    
    y_true_bin = label_binarize(y_true, classes=range(len(class_names)))
    
    plt.figure(figsize=(10, 8))
    
    for i, class_name in enumerate(class_names):
        fpr, tpr, _ = roc_curve(y_true_bin[:, i], y_pred_proba[:, i])
        roc_auc = auc(fpr, tpr)
        plt.plot(fpr, tpr, label=f'{class_name} (AUC = {roc_auc:.2f})')
    
    plt.plot([0, 1], [0, 1], 'k--')
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('ROC Curves')
    plt.legend()
    plt.tight_layout()
    plt.savefig(f'{output_dir}/roc_curves.png', dpi=150)
    plt.show()

def analyze_errors(y_true, y_pred, class_names):
    print("\n" + "=" * 60)
    print("تحليل الأخطاء")
    print("=" * 60)
    
    precision, recall, f1, support = precision_recall_fscore_support(y_true, y_pred)
    
    print(f"\n{'Class':20} {'Precision':10} {'Recall':10} {'F1':10} {'Support':10}")
    print("-" * 60)
    
    for i, class_name in enumerate(class_names):
        print(f"{class_name:20} {precision[i]:10.2%} {recall[i]:10.2%} "
              f"{f1[i]:10.2%} {support[i]:10}")
    
    weak_classes = [(class_names[i], f1[i]) for i in range(len(f1)) if f1[i] < 0.7]
    
    if weak_classes:
        print("\nفئات تحتاج تحسين (F1 < 70%):")
        for name, score in weak_classes:
            print(f"  - {name}: {score:.2%}")

if __name__ == "__main__":
    model = keras.models.load_model('best_model.keras')

        'dataset/test',

        batch_size=32,
        shuffle=False
    )
    class_names = test_ds.class_names
    
    results = evaluate_model(model, test_ds, class_names)

"""
قالب FastAPI للنشر
تشغيل: uvicorn deploy_api:app --host 0.0.0.0 --port 8000 --reload
"""
from fastapi import FastAPI, File, UploadFile, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import JSONResponse
import tensorflow as tf
import numpy as np
from PIL import Image
import io
from typing import List
from pydantic import BaseModel
import time

app = FastAPI(
    title="Image Classifier API",
    description="API لتصنيف الصور باستخدام التعلم العميق",
    version="1.0.0"
)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

MODEL_PATH = "best_model.keras"
CLASS_NAMES = ['Cat', 'Dog', 'Bird']
IMG_SIZE = (224, 224)

model = None

@app.on_event("startup")
async def load_model():
    global model
    model = tf.keras.models.load_model(MODEL_PATH)
    print(f"Model loaded: {MODEL_PATH}")

class Prediction(BaseModel):
    class_name: str
    confidence: float
    inference_time_ms: float

class PredictionResult(BaseModel):
    success: bool
    prediction: Prediction
    all_predictions: List[dict]

async def preprocess_image(file: UploadFile) -> np.ndarray:
    contents = await file.read()
    img = Image.open(io.BytesIO(contents)).convert('RGB')
    img = img.resize(IMG_SIZE)
    img_array = np.array(img) / 255.0
    return np.expand_dims(img_array, 0)

@app.get("/")
async def root():
    return {
        "message": "Image Classifier API",
        "endpoints": {
            "/predict": "POST - Upload image for prediction",
            "/health": "GET - Check API health",
            "/classes": "GET - List available classes"
        }
    }

@app.get("/health")
async def health():
    return {
        "status": "healthy",
        "model_loaded": model is not None
    }

@app.get("/classes")
async def get_classes():
    return {"classes": CLASS_NAMES}

@app.post("/predict", response_model=PredictionResult)
async def predict(image: UploadFile = File(...)):
    if not image.content_type.startswith('image/'):
        raise HTTPException(400, "File must be an image")
    
    start_time = time.time()
    
    try:
        processed = await preprocess_image(image)
        predictions = model.predict(processed, verbose=0)[0]
        
        inference_time = (time.time() - start_time) * 1000
        
        pred_class = np.argmax(predictions)
        
        all_preds = [
            {"class": CLASS_NAMES[i], "probability": float(p)}
            for i, p in enumerate(predictions)
        ]
        all_preds.sort(key=lambda x: x['probability'], reverse=True)
        
        return PredictionResult(
            success=True,
            prediction=Prediction(
                class_name=CLASS_NAMES[pred_class],
                confidence=float(predictions[pred_class]),
                inference_time_ms=inference_time
            ),
            all_predictions=all_preds
        )
        
    except Exception as e:
        raise HTTPException(500, f"Prediction failed: {str(e)}")

@app.post("/predict/batch")
async def predict_batch(images: List[UploadFile] = File(...)):
    if len(images) > 10:
        raise HTTPException(400, "Maximum 10 images per batch")
    
    results = []
    for image in images:
        try:
            processed = await preprocess_image(image)
            predictions = model.predict(processed, verbose=0)[0]
            pred_class = np.argmax(predictions)
            
            results.append({
                "filename": image.filename,
                "class": CLASS_NAMES[pred_class],
                "confidence": float(predictions[pred_class])
            })
        except Exception as e:
            results.append({
                "filename": image.filename,
                "error": str(e)
            })
    
    return {"results": results}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
