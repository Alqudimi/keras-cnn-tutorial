نشر النموذج (Deployment)

تحويل نموذجك المدرب إلى تطبيق فعلي يمكن للمستخدمين استخدامه.

مقدمة

بعد تدريب النموذج، تأتي مرحلة النشر - وهي الخطوة الأهم لتحويل عملك إلى قيمة فعلية. هناك عدة طرق للنشر حسب الاستخدام المستهدف.

طريقة النشر الاستخدام المزايا
TensorFlow Lite الهواتف، IoT خفيف، سريع، يعمل بدون إنترنت
TensorFlow.js المتصفح لا يحتاج خادم، خصوصية عالية
Flask/FastAPI خادم API مرونة عالية، سهل التطوير
ONNX منصات متعددة محمول بين الأطر المختلفة

import tensorflow as tf

# التحويل الأساسي
converter = tf.lite.TFLiteConverter.from_saved_model('my_model')
tflite_model = converter.convert()

# حفظ النموذج
with open('model.tflite', 'wb') as f:
    f.write(tflite_model)

# التحويل مع التحسينات (Quantization)
converter = tf.lite.TFLiteConverter.from_saved_model('my_model')

# Dynamic Range Quantization (أسرع، حجم أصغر)
converter.optimizations = [tf.lite.Optimize.DEFAULT]

tflite_quant_model = converter.convert()
with open('model_quantized.tflite', 'wb') as f:
    f.write(tflite_quant_model)

# Full Integer Quantization (الأسرع)
def representative_dataset():
    for img_path in sample_images[:100]:
        img = tf.io.read_file(img_path)
        img = tf.image.decode_jpeg(img, channels=3)
        img = tf.image.resize(img, [224, 224])
        img = tf.expand_dims(img / 255.0, 0)
        yield [tf.cast(img, tf.float32)]

converter = tf.lite.TFLiteConverter.from_saved_model('my_model')
converter.optimizations = [tf.lite.Optimize.DEFAULT]
converter.representative_dataset = representative_dataset
converter.target_spec.supported_types = [tf.int8]
converter.inference_input_type = tf.uint8
converter.inference_output_type = tf.uint8

tflite_int8_model = converter.convert()
with open('model_int8.tflite', 'wb') as f:
    f.write(tflite_int8_model)

# اختبار النموذج
interpreter = tf.lite.Interpreter(model_path='model.tflite')
interpreter.allocate_tensors()

input_details = interpreter.get_input_details()
output_details = interpreter.get_output_details()

# التنبؤ
interpreter.set_tensor(input_details[0]['index'], input_data)
interpreter.invoke()
predictions = interpreter.get_tensor(output_details[0]['index'])

        let model;
        const classNames = ['Cat', 'Dog', 'Bird'];
        
        // تحميل النموذج
        async function loadModel() {
            model = await tf.loadLayersModel('tfjs_model/model.json');
            console.log('Model loaded!');
        }
        
        // التنبؤ
        async function predict(imageElement) {
            // تحويل الصورة إلى tensor
            const tensor = tf.browser.fromPixels(imageElement)
                .resizeNearestNeighbor([224, 224])
                .toFloat()
                .div(255.0)
                .expandDims();
            
            // التنبؤ
            const predictions = await model.predict(tensor).data();
            
            // الحصول على أعلى تنبؤ
            const maxIndex = predictions.indexOf(Math.max(...predictions));
            const confidence = predictions[maxIndex] * 100;

                `${classNames[maxIndex]}: ${confidence.toFixed(1)}%`;
            
            // تنظيف الذاكرة
            tensor.dispose();
        }

            const file = e.target.files[0];
            const img = document.getElementById('preview');
            img.src = URL.createObjectURL(file);
            img.onload = () => predict(img);
        });

Flask API

Python - app.py 
from flask import Flask, request, jsonify
import tensorflow as tf
import numpy as np
from PIL import Image
import io

app = Flask(__name__)

# تحميل النموذج عند بدء التطبيق
model = tf.keras.models.load_model('my_model')
class_names = ['Cat', 'Dog', 'Bird']

def preprocess_image(image_bytes):
    """معالجة الصورة"""
    img = Image.open(io.BytesIO(image_bytes))
    img = img.resize((224, 224))
    img = img.convert('RGB')
    img_array = np.array(img) / 255.0
    return np.expand_dims(img_array, 0)

@app.route('/predict', methods=['POST'])
def predict():
    if 'image' not in request.files:
        return jsonify({'error': 'No image provided'}), 400
    
    file = request.files['image']
    image_bytes = file.read()
    
    # المعالجة والتنبؤ
    processed_image = preprocess_image(image_bytes)
    predictions = model.predict(processed_image)[0]
    
    # النتائج
    results = []
    for i, prob in enumerate(predictions):
        results.append({
            'class': class_names[i],
            'probability': float(prob)
        })
    
    # ترتيب حسب الاحتمالية
    results.sort(key=lambda x: x['probability'], reverse=True)
    
    return jsonify({
        'prediction': results[0]['class'],
        'confidence': results[0]['probability'],
        'all_predictions': results
    })

@app.route('/health', methods=['GET'])
def health():
    return jsonify({'status': 'healthy'})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)

FastAPI (أسرع وأحدث)

Python - main.py 
from fastapi import FastAPI, File, UploadFile, HTTPException
from fastapi.responses import JSONResponse
import tensorflow as tf
import numpy as np
from PIL import Image
import io
from typing import List
from pydantic import BaseModel

app = FastAPI(title="Image Classifier API")

# تحميل النموذج
model = tf.keras.models.load_model('my_model')
class_names = ['Cat', 'Dog', 'Bird']

class PredictionResult(BaseModel):
    prediction: str
    confidence: float
    all_predictions: List[dict]

async def preprocess_image(file: UploadFile) -> np.ndarray:
    contents = await file.read()
    img = Image.open(io.BytesIO(contents))
    img = img.resize((224, 224)).convert('RGB')
    return np.expand_dims(np.array(img) / 255.0, 0)

@app.post("/predict", response_model=PredictionResult)
async def predict(image: UploadFile = File(...)):
    if not image.content_type.startswith('image/'):
        raise HTTPException(400, "File must be an image")
    
    processed = await preprocess_image(image)
    predictions = model.predict(processed)[0]
    
    results = [
        {"class": class_names[i], "probability": float(p)}
        for i, p in enumerate(predictions)
    ]
    results.sort(key=lambda x: x['probability'], reverse=True)
    
    return PredictionResult(
        prediction=results[0]['class'],
        confidence=results[0]['probability'],
        all_predictions=results
    )

@app.get("/health")
async def health():
    return {"status": "healthy"}

# تشغيل: uvicorn main:app --reload

ONNX (Open Neural Network Exchange)

ONNX يسمح بتشغيل النموذج على منصات مختلفة (PyTorch, TensorFlow, تطبيقات C++, إلخ).

Python 
# pip install tf2onnx onnxruntime

import tf2onnx
import onnx
import onnxruntime as ort
import numpy as np

# التحويل إلى ONNX
model = tf.keras.models.load_model('my_model')

# تحويل
spec = (tf.TensorSpec((None, 224, 224, 3), tf.float32, name="input"),)
model_proto, _ = tf2onnx.convert.from_keras(model, input_signature=spec)

# حفظ
onnx.save_model(model_proto, 'model.onnx')

# التحقق
onnx_model = onnx.load('model.onnx')
onnx.checker.check_model(onnx_model)
print("ONNX model is valid!")

# الاستخدام مع ONNX Runtime
session = ort.InferenceSession('model.onnx')

# التنبؤ
input_name = session.get_inputs()[0].name
output_name = session.get_outputs()[0].name

predictions = session.run(
    [output_name],
    {input_name: input_data.astype(np.float32)}
)[0]

print(f"Prediction: {class_names[np.argmax(predictions)]}")
