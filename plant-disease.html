def focal_loss(gamma=2., alpha=0.25):
    """
    Focal Loss - ูุฑูุฒ ุนูู ุงูุฃูุซูุฉ ุงูุตุนุจุฉ
    """
    def focal_loss_fn(y_true, y_pred):
        y_pred = tf.clip_by_value(y_pred, 1e-7, 1 - 1e-7)
        cross_entropy = -y_true * tf.math.log(y_pred)
        weight = alpha * y_true * tf.pow(1 - y_pred, gamma)
        return tf.reduce_sum(weight * cross_entropy, axis=-1)

# ุทุจูุงุช Augmentation ุงููููุฉ
class StrongAugmentation(tf.keras.layers.Layer):
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        
        self.augment_layers = tf.keras.Sequential([
            layers.RandomFlip("horizontal_and_vertical"),
            layers.RandomRotation(0.4),
            layers.RandomZoom((-0.3, 0.3)),
            layers.RandomTranslation(0.2, 0.2),
            layers.RandomContrast(0.3),
            layers.RandomBrightness(0.3),
        ])
    
    def call(self, images, training=True):
        if training:
            return self.augment_layers(images)
        return images

# Cutout / Random Erasing
class RandomCutout(tf.keras.layers.Layer):
    def __init__(self, probability=0.5, min_size=0.1, max_size=0.3):
        super().__init__()
        self.probability = probability
        self.min_size = min_size
        self.max_size = max_size
    
    def call(self, images, training=True):
        if not training:
            return images
        
        if tf.random.uniform([]) > self.probability:
            return images
        
        batch_size = tf.shape(images)[0]
        height = tf.shape(images)[1]
        width = tf.shape(images)[2]
        
        # Random cutout size
        cutout_h = tf.cast(tf.cast(height, tf.float32) * 
                          tf.random.uniform([], self.min_size, self.max_size), tf.int32)
        cutout_w = tf.cast(tf.cast(width, tf.float32) * 
                          tf.random.uniform([], self.min_size, self.max_size), tf.int32)
        
        # Random position
        y = tf.random.uniform([], 0, height - cutout_h, dtype=tf.int32)
        x = tf.random.uniform([], 0, width - cutout_w, dtype=tf.int32)
        
        # Create mask
        mask = tf.ones((batch_size, height, width, tf.shape(images)[3]))
        mask = mask - tf.pad(
            tf.ones((batch_size, cutout_h, cutout_w, tf.shape(images)[3])),
            [[0, 0], [y, height - y - cutout_h], [x, width - x - cutout_w], [0, 0]]
        )
        
        return images * mask

# MixUp Augmentation
def mixup(images, labels, alpha=0.2):
    """
    MixUp: ุฎูุท ุงูุตูุฑ ูุงูุชุณููุงุช
    """
    batch_size = tf.shape(images)[0]
    
    # Sample lambda from Beta distribution
    lam = tf.random.uniform([], 0, alpha)
    
    # Random permutation
    indices = tf.random.shuffle(tf.range(batch_size))
    
    # Mix images and labels
    mixed_images = lam * images + (1 - lam) * tf.gather(images, indices)
    mixed_labels = lam * labels + (1 - lam) * tf.gather(labels, indices)
    
    return mixed_images, mixed_labels

def residual_block(x, filters, stride=1, use_projection=False):
    """
    Residual Block ูุน skip connection
    """
    shortcut = x
    
    # First conv
    x = Conv2D(filters, 3, strides=stride, padding='same',
               kernel_regularizer=l2(1e-4))(x)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    
    # Second conv
    x = Conv2D(filters, 3, padding='same',
               kernel_regularizer=l2(1e-4))(x)
    x = BatchNormalization()(x)
    
    # Projection shortcut
    if use_projection or stride != 1:
        shortcut = Conv2D(filters, 1, strides=stride, padding='same')(shortcut)
        shortcut = BatchNormalization()(shortcut)
    
    x = Add()([x, shortcut])
    x = Activation('relu')(x)
    
    return x

def build_plant_disease_model(input_shape=(224, 224, 3), num_classes=38):
    """
    ูููุฐุฌ ุนููู ูุงูุชุดุงู ุฃูุฑุงุถ ุงููุจุงุชุงุช
    """
    inputs = Input(shape=input_shape)
    
    # Stem
    x = Conv2D(64, 7, strides=2, padding='same', kernel_regularizer=l2(1e-4))(inputs)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    x = MaxPooling2D(3, strides=2, padding='same')(x)
    
    # Stage 1: 64 filters
    x = residual_block(x, 64, use_projection=True)
    x = residual_block(x, 64)
    x = Dropout(0.2)(x)
    
    # Stage 2: 128 filters
    x = residual_block(x, 128, stride=2, use_projection=True)
    x = residual_block(x, 128)
    x = residual_block(x, 128)
    x = Dropout(0.3)(x)
    
    # Stage 3: 256 filters
    x = residual_block(x, 256, stride=2, use_projection=True)
    x = residual_block(x, 256)
    x = residual_block(x, 256)
    x = residual_block(x, 256)
    x = Dropout(0.4)(x)
    
    # Stage 4: 512 filters
    x = residual_block(x, 512, stride=2, use_projection=True)
    x = residual_block(x, 512)
    x = Dropout(0.4)(x)
    
    # Head
    x = GlobalAveragePooling2D()(x)
    x = Dense(1024, activation='relu', kernel_regularizer=l2(1e-4))(x)
    x = BatchNormalization()(x)
    x = Dropout(0.5)(x)
    x = Dense(512, activation='relu', kernel_regularizer=l2(1e-4))(x)
    x = Dropout(0.4)(x)
    outputs = Dense(num_classes, activation='softmax')(x)
    
    model = Model(inputs, outputs, name='plant_disease_detector')
    return model

def cosine_annealing_schedule(epoch, lr, total_epochs=100, min_lr=1e-6):
    """
    Cosine Annealing scheduler
    """
    return min_lr + 0.5 * (lr - min_lr) * (1 + np.cos(np.pi * epoch / total_epochs))
